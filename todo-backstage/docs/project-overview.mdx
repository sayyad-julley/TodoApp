### Code Organization

```javascript
// Component pattern example
import React, { useState, useEffect } from 'react';
import { useQuery } from 'react-query';

const TodoList = () => {
  const [todos, setTodos] = useState([]);

  const { data, isLoading, error } = useQuery(
    'todos',
    async () => {
      const response = await fetch('/api/todos');
      return response.json();
    }
  );

  // Error handling with boundaries
  if (error) {
    return <ErrorBoundary error={error} />;
  }

  // Loading states
  if (isLoading) {
    return <LoadingSpinner />;
  }

  return (
    <div className="todo-list">
      {data?.map(todo => (
        <TodoItem key={todo.id} todo={todo} />
      ))}
    </div>
  );
};
```

### API Design Patterns

```javascript
// RESTful API pattern with proper error handling
const router = express.Router();

// Input validation middleware
router.post('/todos',
  validateTodoInput,
  async (req, res, next) => {
    try {
      const todo = await TodoService.create(req.body);
      res.status(201).json({
        success: true,
        data: todo,
        message: 'Todo created successfully'
      });
    } catch (error) {
      next(error);
    }
  }
);

// Error handling middleware
const errorHandler = (error, req, res, next) => {
  const status = error.status || 500;
  const message = error.message || 'Internal server error';

  res.status(status).json({
    success: false,
    error: {
      message,
      ...(process.env.NODE_ENV === 'development' && { stack: error.stack })
    }
  });
};
```

## Local Development

### Prerequisites

- Node.js 18+ and npm
- Docker and Docker Compose
- MongoDB (local or Atlas)
- AWS CLI (for infrastructure management)

### Setup Instructions

<Steps>
  <Step title="Clone Repository">
    ```bash
    git clone https://github.com/sayyad-julley/TodoApp
    cd TodoApp-GoldenPath
    ```
  </Step>
  <Step title="Start Local Services">
    ```bash
    # Start MongoDB and other services
    docker-compose up -d

    # Install dependencies
    npm install
    ```
  </Step>
  <Step title="Configure Environment">
    ```bash
    cp .env.example .env
    # Edit .env with your configuration
    ```
  </Step>
  <Step title="Start Development Servers">
    ```bash
    # Start backend (port 7007)
    npm run dev:backend

    # Start frontend (port 3000)
    npm run dev:frontend
    ```
  </Step>
</Steps>
